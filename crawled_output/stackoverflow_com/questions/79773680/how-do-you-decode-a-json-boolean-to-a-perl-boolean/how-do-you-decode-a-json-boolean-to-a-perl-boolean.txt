Since 5.40, you can use Cpanel::JSON::XS->new->unblessed_bool->decode instead of decode_json.
use v5.40;
BEGIN {
no warnings qw( once experimental );
*is_bool = sub { builtin::is_bool( $_[0] ) };
}
use Cpanel::JSON::XS qw( );
use Devel::Peek
qw( Dump );
say Cpanel::JSON::XS->VERSION;
my $j = Cpanel::JSON::XS->new->unblessed_bool->allow_nonref;
my $orig_data = true;
say is_bool( $orig_data ) ? "boolean" : "not boolean";
Dump( $orig_data );
my $json = $j->encode( $orig_data );
say $json;
my $obtained_data = $j->decode( $json );
say is_bool( $obtained_data ) ? "boolean" : "not boolean";
Dump( $obtained_data );
