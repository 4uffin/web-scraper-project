How to detect if a widget starts under the AppBar or below it in Flutter?

I have a Scaffold with extendBodyBehindAppBar: true because I want a blurred AppBar.
The problem is that sometimes my body widget (inner) starts behind the AppBar instead of below it.
Here is a simplified version of my code:
import 'dart:ui';
import 'package:app/src/core/extension/widget_extension.dart';
import 'package:app/src/presentation/bloc/base/base_bloc.dart';
import 'package:flutter/material.dart';
import 'package:flutter_bloc/flutter_bloc.dart';
abstract class AdaptivePage extends StatefulWidget {
const AdaptivePage({super.key});
@override
State<AdaptivePage> createState();
}
abstract class AdaptivePageState<
T extends AdaptivePage,
B extends BaseBloc<BaseEvent, S>,
S extends BaseState
>
extends State<T>
with AutomaticKeepAliveClientMixin {
Widget buildBody(BuildContext context);
PreferredSizeWidget? buildAppBar(BuildContext context) => null;
Widget? buildBottomNavigationBar(BuildContext context) => null;
Widget? buildFloatingActionButton(BuildContext context) => null;
List<BlocListener<B, S>> blocListeners() => [];
BlocListener<B, S> errorListener() {
return BlocListener<B, S>(
listenWhen: (prev, curr) => prev.errorMessage != curr.errorMessage && curr.hasError,
listener: (context, state) {
state.errorMessage!.showError(context);
},
);
}
B? createBloc() => null;
@override
Widget build(BuildContext context) {
super.build(context);
Widget inner = buildBody(context);
if (inner is! ScrollView) {
final topPadding = (buildAppBar(context) != null
? kToolbarHeight + MediaQuery.of(context).padding.top
: 0.0);
inner = Padding(
padding: EdgeInsets.only(top: topPadding),
child: inner,
);
}
final B? bloc = createBloc();
final List<BlocListener<B, S>> listeners = [...blocListeners()];
listeners.add(errorListener());
Widget bodyWidget;
if (bloc != null) {
bodyWidget = BlocProvider<B>(
create: (_) => bloc,
child: Builder(
builder: (context) {
return listeners.isNotEmpty
? MultiBlocListener(listeners: listeners, child: inner)
: inner;
},
),
);
} else {
bodyWidget = listeners.isNotEmpty
? MultiBlocListener(listeners: listeners, child: inner)
: inner;
}
return Scaffold(
extendBodyBehindAppBar: true,
appBar: buildAppBar(context) != null
? PreferredSize(
preferredSize: Size.fromHeight(context.isIOS ? 44 : kToolbarHeight),
child: ClipRRect(
child: BackdropFilter(
filter: ImageFilter.blur(sigmaX: 10, sigmaY: 10),
child: buildAppBar(context),
),
),
)
: null,
body: bodyWidget,
bottomNavigationBar: buildBottomNavigationBar(context),
floatingActionButton: buildFloatingActionButton(context),
);
}
@override
bool get wantKeepAlive => false;
}
What I want:
If the body already starts below the AppBar, do nothing.
If the body starts behind the AppBar, add topPadding so that it moves down.
Things I tried:
Checking inner is ScrollView (doesn’t work, because the ScrollView may be nested inside other widgets).
Wrapping in a LayoutBuilder and measuring position with RenderBox.localToGlobal (didn’t work properly with ListView, also broke gesture events).
How can I reliably detect whether my body widget is rendered under the AppBar or below it, so that I can add padding only when needed?
