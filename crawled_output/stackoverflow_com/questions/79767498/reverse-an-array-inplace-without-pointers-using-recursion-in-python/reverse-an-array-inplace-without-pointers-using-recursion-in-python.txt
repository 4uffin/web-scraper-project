class Solution:
def reverse(self, arr: list, n: int) -> None:
if len(arr)-n>=n-1:
return
arr[len(arr)-n],arr[n-1]=arr[n-1],arr[len(arr)-n]
self.reverse(arr,n-1)
return arr
s=Solution()
print(s.reverse([1,2,3,4,5,6,7],7))
