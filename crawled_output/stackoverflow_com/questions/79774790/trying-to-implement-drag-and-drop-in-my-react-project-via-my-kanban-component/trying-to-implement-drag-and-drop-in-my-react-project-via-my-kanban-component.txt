as part of a group project, I am trying to set up a kanban system similar to those used by YouTrack, Trello, and others to manage project tasks from one status to another by moving them from one column to another. To do this, I started by using << dnd-kit/core >> and its presets before discovering << dnd-kit/react >>, and more recently << Pragmatic drag and drop >>. However, I am not getting the desired result. Either the tasks are duplicated when I use the “for” loop, or there is no drag and drop, so I don't know which one would be best suited to my case between core, react, and pragmatic, whether I should change or continue with core. I am sharing the contents of these two files (Board and config) with you and, if necessary, I can provide you with more details.
kanban view

and my board:
import { useState, useCallback, useMemo } from "react";
import {
DndContext,
PointerSensor,
useSensor,
useSensors,
type DragStartEvent,
type DragEndEvent,
DragOverlay,
closestCorners,
} from "@dnd-kit/core"
import ColumnContainer from "@/components/projectLanding/kanbanComp/KanbanColumn"
import TaskCard from "@/components/projectLanding/kanbanComp/KanbanTask"
import type { ITask, KanbanProps, Status } from "@/services/interfaces/kanbanConfig"
/**
* KanbanBoard:
* - Manages the Kanban board with drag & drop
* - Detects movements between columns or within the same column
* - Uses status as key to organize tasks
*/
const KanbanBoard = ({
columns,
tasks,
onTaskMove,
onTaskUpdate,
onTaskDelete,
className,
}: KanbanProps) => {
const sensors = useSensors(
useSensor(PointerSensor, {
activationConstraint: {
distance: 10,
},
})
);
// State to track the currently active (dragged) task
const [activeTask, setActiveTask] = useState<ITask | null>(null);
// Memoized accessor to get tasks by status and keep stable ref between renders
const tasksByStatus = useMemo(
() => (s: Status) => tasks[s] || [],
[tasks]
);
/**
* Handle drag start event
* - Retrieves the active task data from the useSortable in the TaskCard
* - Sets the active task state to show in the DragOverlay
*/
const handleDragStart = useCallback((event: DragStartEvent) => {
const { active } = event;
// Get the active task data from the useSortable in the TaskCard
const task = active.data.current?.task as ITask | undefined;
if (task) setActiveTask(task);
}, []);
/**
* Handle drag end event
* - Determines source and target columns and indices
* - Calls onTaskMove if the task was moved to a different position or column
* - Resets the active task state to hide the DragOverlay
*/
const handleDragEnd = useCallback((event: DragEndEvent) => {
const { active, over } = event;
setActiveTask(null);
if (!over) {
console.log("Aucune cible n'a été détectée, le drag est annulé");
return;
}
console.log(" Vérification des datas : ", {
activeData: active.data.current,
activeId: active.id,
overData: over.data.current,
overId: over.id,
});
const taskId = active.id as number;
let from = active.data.current?.container as Status | undefined;
for (const status in tasks) {
if (tasks[status as Status].some(task => task.id === taskId)) {
from = status as Status;
break;
}
}
if (!from) {
console.error("Impossible de déterminer la colonne source du déplacement");
return;
}
const oldIndex = active.data.current?.index as number | undefined;
const to = over.data.current?.container as Status | undefined;
if (!from || !to) return;
let toIndex: number | undefined;
if (over.data.current?.type === "column") {
const targetList = tasks[to] || [];
toIndex = targetList.length;
} else {
// Dropped on a task; compute index in target list
const targetList = tasks[to] || [];
const overId = over.id as number;
// find index of the target element
toIndex = targetList.findIndex((t) => t.id === overId);
if (toIndex === -1) {
// If not found, it means we dropped after the last item
toIndex = targetList.length;
}
}
// If moving within the same column and index didn't change, do nothing
if (from === to) {
if (
typeof oldIndex === "number" &&
typeof toIndex === "number" &&
oldIndex === toIndex
) {
return;
}
}
// Call the move handler to update state in parent
onTaskMove(taskId, from, to, toIndex);
},
[onTaskMove, tasks]
);
return (
<DndContext
sensors={sensors}
collisionDetection={closestCorners}
onDragStart={handleDragStart}
onDragEnd={handleDragEnd}
>
<div className={`flex gap-6 overflow-x-auto ${className}`}>
{columns.map((col) => (
<ColumnContainer
key={col.id}
column={col}
tasks={tasksByStatus(col.status)}
onTaskEdit={onTaskUpdate}
onTaskDelete={onTaskDelete}
activeTaskId={activeTask?.id}
/>
))}
</div>
<DragOverlay>
{activeTask ? (
<div className="z-50 w-72">
<TaskCard
task={activeTask}
columnStatus={activeTask.status}
index={0}
onEdit={() => { }}
onDelete={() => { }}
/>
</div>
) : null}
</DragOverlay>
</DndContext>
)
}
export default KanbanBoard
