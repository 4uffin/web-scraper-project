update-score.py
import csv
import concurrent.futures
import urllib.request
import re
FNAME = "CS 분야 우수 학술대회 목록.csv"
head = []
rows = []
scores = {
1: {"최우수": 1, "우수": 0.5, "": 0},
2: {"4": 1, "3": 0.75, "2": 0.5, "1": 0.25, "": 0},
3: {"O": 1, "": 0},
4: {"O": 1, "": 0},
5: {"최우수": 1, "우수": 0.5, "": 0},
}
with open(FNAME, "r") as csvfile:
reader = csv.reader(csvfile)
head = next(reader)
for row in reader:
row_scores = []
for idx in scores:
score = scores[idx][row[idx]]
row_scores.append(score)
row[6] = "%.2f" % (sum(row_scores) / len(row_scores))
rows.append(row)
rows = sorted(rows, key=lambda x: x[0].lower())
def update_name(row):
try:
with urllib.request.urlopen(
"https://dblp.org/db/%s/index.html" % row[8]
) as response:
html = response.read().decode("utf8")
m = re.search(r"<h1>(.*?)</h1>", html)
name = m.group(1).replace("&amp;", "&")
for v in ("(poster)", "(spotlight)", "(oral)"):
if row[0].endswith(v):
name += " " + v
break
print(name)
row[7] = name
except urllib.error.HTTPError as E:
print("ERR", E)
pass
# with concurrent.futures.ThreadPoolExecutor() as executor:
#
futures = [executor.submit(update_name, row) for row in rows]
#
concurrent.futures.wait(futures)
with open(FNAME, "w") as csvfile:
writer = csv.writer(csvfile)
writer.writerow(head)
writer.writerows(rows)
