Connect-CIServerV2.ps1
<#
.SYNOPSIS
Adapted version of the Connect-CIServer Cmdlet to connect to VMware Cloud Director
with an API Access Token
.DESCRIPTION
Adapted version of the Connect-CIServer Cmdlet to connect to VMware Cloud Director
with an API Access Token.
This Cmdlet 'wraps' the original Connect-CIServer and simplifies the REST API calls needed to
'rehydrate' a PowerCLI Session from an API Access Token Generated in VCD.
.PARAMETER ApiToken
Access Token. Fore more info, please check:
https://blogs.vmware.com/cloudprovider/2022/03/cloud-director-api-token.html
.PARAMETER Server
VCD Host name.
.PARAMETER Org
Name of the Org to log into. Defaults to 'System' if not provided.
.SYNTAX
Connect-CIServer [-Server] <String> [-ApiToken] <String> [-Org] <String>
.NOTES
For more info on the original 'Connect-CIServer', please check:
http://developer.vmware.com/docs/powercli/latest/vmware.vimautomation.cloud/commands/connect-ciserver
#>
function Connect-CIServerV2 {
param (
[Parameter (Mandatory = $true)][String]$ApiToken,
[Parameter (Mandatory = $true)][String]$Server,
[Parameter (Mandatory = $false)][String]$Org="System"
)
if ($Org.ToLower() -eq "system") {
$uri = "https://$($Server)/oauth/provider/token"
} else {
$uri = "https://$($Server)/oauth/tenant/$($Org)/token"
}
$body = "grant_type=refresh_token&refresh_token=$($ApiToken)"
$headers = @{
'Accept' = 'application/json'
'ContentType' = 'application/x-www-form-urlencoded'
}
try {
$access_token = (Invoke-RestMethod -Method Post -Uri $uri -Headers $headers -Body $body).access_token
Write-Host -ForegroundColor Green ("Created access_token from token successfully")
} catch {
Write-Host -ForegroundColor Red ("Could not create access_token from token, response code: $(Exception.Response.StatusCode.value__)")
Write-Error ("Status Description: $($_.Exception).")
break
}
try {
Connect-CIServer -Server $Server -SessionId "Bearer $access_token"
} catch {
Write-Host -ForegroundColor Red ("Could not connect to VCD, response code: $(Exception.Response.StatusCode.value__)")
Write-Error ("Status Description: $($_.Exception).")
}
}
