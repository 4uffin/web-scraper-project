#!/bin/bash
# Find all top-level node_modules directories and calculate their total size
total_size=0
for dir in $(find . -name "node_modules" -type d -not -path "*/node_modules/*"); do
size=$(du -sk "$dir" | cut -f1)
total_size=$(($total_size + $size))
done
